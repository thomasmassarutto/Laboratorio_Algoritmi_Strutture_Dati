import java.util.Scanner;
import java.util.*;
public class ES22 {



    public static void resolveRequest(String[] command, AVL tree){

        switch (command[0]){
            case "insert":
                int KeyVal = Integer.parseInt(command[1]);
                String KeyString = command[2];
                tree.root = tree.insert(tree.root, KeyVal, KeyString);
                return;
            case "remove":
                int KeyValDel = Integer.parseInt(command[1]);
                tree.root = tree.deleteNode(tree.root, KeyValDel);
                return;
            case "find":
                tree.find(tree.root, Integer.parseInt(command[1]));
                return;
            case "show":
                tree.show(tree.root);
                return;
            case "exit":
                System.exit(0);
            default:
                System.exit(0);
        }
    }

    public static void main(String[] args) {
        int k=1;
        AVL tree = new AVL();
        Scanner stringa = new Scanner(System.in);
        do {
            
            String s = stringa.nextLine();
            String[] command = s.split(" ");
            resolveRequest(command, tree);
        }while(stringa.hasNext());
    }
}
